def extract_id(raw_str):
    if not raw_str: return None
    match = re.search(r'\[id(\d+)\|', raw_str)
    if match: return int(match.group(1))
    nums = re.findall(r'\d+', raw_str)
    if nums: return int(nums[0])
    return None

def fmt_big(amount):
    if amount is None: return "0"
    if amount >= 1000000:
        return f"{amount/1000000:.1f}M"
    elif amount >= 1000:
        return f"{int(amount)}" if amount == int(amount) else f"{amount:.2f}"
    return f"{int(amount)}" if amount == int(amount) else f"{amount:.2f}".rstrip('0').rstrip('.')

def fmt_time(td):
    total_seconds = int(td.total_seconds())
    if total_seconds < 60:
        return f"{total_seconds} сек"
    elif total_seconds < 3600:
        return f"{total_seconds // 60} мин {total_seconds % 60} сек"
    else:
        hours = total_seconds // 3600
        minutes = (total_seconds % 3600) // 60
        return f"{hours} ч {minutes} мин"

def can_get_bonus(uid):
    user = get_user(uid)
    if not user or not user['last_bonus']:
        return True, None
    last = datetime.strptime(user['last_bonus'], "%Y-%m-%d %H:%M:%S")
    delta = timedelta(hours=3)
    if datetime.now() - last >= delta:
        return True, None
    else:
        next_time = last + delta
        remaining = next_time - datetime.now()
        return False, remaining

def add_car(uid, car_name):

            send(api, uid, f"✅ Пользователь исключён из беседы.", peer_id)
        except Exception as e:
            if "9" in str(e):
                send(api, uid, "❌ У бота нет прав администратора в этой беседе.", peer_id)
            elif "15" in str(e):
                send(api, uid, "❌ Нельзя кикнуть администратора беседы.", peer_id)
            else:
                send(api, uid, f"❌ Ошибка", peer_id)
        return

    # ========== ВЫДАЧА ДЕНЕГ (АДМИН) ==========
    if cmd == "givemoney" and len(parts) >= 3:
        if uid not in ADMIN_IDS:
            send(api, uid, "⛔ У вас нет прав.", peer_id)
            return
        try:
            target = int(parts[1])
            amount = float(parts[2].replace(',', '.'))
        except:
            send(api, uid, "❌ Формат: /givemoney [id] [сумма]", peer_id)
            return
        if amount <= 0:
            send(api, uid, "❌ Сумма должна быть больше 0", peer_id)
            return
        target_user = get_user(target)
        if not target_user:
            send(api, uid, f"❌ Пользователь с ID {target} не найден", peer_id)
            return
        update_balance(target, amount)
        send(api, uid, f"✅ Выдали {fmt_big(amount)}$ пользователю {target_user['username']}", peer_id)
        return

    # ========== ТОП БОГАЧЕЙ ==========
    if cmd == "топ" or cmd == "top":
        with sqlite3.connect(DB_NAME) as c:
            c.row_factory = sqlite3.Row
            top = c.execute("SELECT username, balance FROM users ORDER BY balance DESC LIMIT 10").fetchall()
        if not top:
            send(api, uid, "📊 Топ богачей пока пуст", peer_id)
            return
        msg = "💰 ТОП-10 БОГАЧЕЙ 💰\n\n"
        for i, u in enumerate(top, 1):
            msg += f"{i}. {u['username']} — {fmt_big(u['balance'])}$\n"
        send(api, uid, msg, peer_id)
        return

    # ========== ПОМОЩЬ ==========
    if cmd in ["help", "команды"]:
        help_text = (
            "🎮 КОМАНДЫ БОТА 🎮\n\n"
            "👤 Профиль: /меню\n\n"
            "🎁 Бонус: /бонус (раз в 3 часа, 50000-100000$)\n\n"
            "🚗 Машины: /машины — список машин\n"
            "🚗 Покупка: /buycar [номер] — купить машину\n\n"
            "🏦 Финансы:\n/банк положить [сумма]\n/банк снять [сумма]\n/топ — топ богачей\n\n"
            "🛒 Магазин: /shop, /buy [товар] [кол-во]\n\n"
            "🎰 Казино: /казино [сумма]\n\n"
            "👑 Админ: /кик, /givemoney"
        )
        send(api, uid, help_text, peer_id)
        return

    logger.info("Запуск бота...")

    while True:
        try:
            vk = vk_session.get_api()
            logger.info("✅ Авторизация успешна!")
            longpoll = VkBotLongPoll(vk_session, GROUP_ID)
            logger.info("✅ Бот работает! Напиши /меню")
            for event in longpoll.listen():
                if event.type == VkBotEventType.MESSAGE_NEW:
                    handle_message(vk, event)

            time.sleep(5)
            vk_session = VkApi(token=VK_TOKEN, api_version="5.131")

if __name__ == "__main__":
